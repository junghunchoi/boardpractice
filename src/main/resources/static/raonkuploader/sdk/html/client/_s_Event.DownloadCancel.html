K Upload :: Event.DownloadCancel

Event.DownloadCancel

callback 이벤트를 설정합니다.

remarks

이벤트 RAONKUPLOAD_DownloadCancel를 설정합니다.
다운로드 중단 시 이벤트가 발생합니다.
Event Object로 Event들을 모아서 설정이 가능합니다.

sample code

<script type="text/javascript" src="raonkupload/js/raonkupload.js"></script>	

<!-- ..... 생략 ..... -->
 
<div style="width:800px;height:200px">  
    <script type="text/javascript">
        // 이벤트발생시 호출할 함수를 설정합니다.
        var eventFunc = function(eventParam1, eventParam2, eventParam3){
            alert("Event!");
            // work something ...
        }

        // 다운로드 중단 시 이벤트가 발생합니다.
        var raonkParam = {
            .....
            Event: {
                DownloadCancel : eventFunc, 
                // Add Another Event Config ...
            }
        }

        new RAONKUpload(raonkParam);
    </script>
</div>
