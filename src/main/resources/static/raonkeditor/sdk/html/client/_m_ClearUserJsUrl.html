K Editor :: ClearUserJsUrl

void ClearUserJsUrl(editorName)

addUserJsUrl로 추가된 script를 모두 제거합니다.

return value

없음.

parameters

editorName  에디터의 ID를 의미합니다.

remarks

없음.

sample code

<!-- ..... 생략 ..... -->

<script type="text/javascript" src="raonkeditor/js/raonkeditor.js"></script>
<script type="text/javascript">
    function clearUserJsUrl() {
        var url = 'http://www.raonk.com/raonkeditor/js/test.js'
        RAONKEDITOR.addUserJsUrl(url, 'editor1');
        // id가 editor1인 에디터의 편집영역 iframe document의 head 영역에 addUserJsUrl로 추가된 script를 모두 제거합니다.
        RAONKEDITOR.clearUserJsUrl('editor1');
    }
</script>	

<!-- ..... 생략 ..... -->
 
<div style="width:900px;height:550px">
    <script type="text/javascript">
        new RAONKEditor({Id:'editor1'});
    </script>
</div>
